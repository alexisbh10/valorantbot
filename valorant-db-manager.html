from fastapi import FastAPI, Request
from fastapi.middleware.cors import CORSMiddleware
import asyncpg

app = FastAPI()
app.add_middleware(CORSMiddleware, allow_origins=["*"], allow_methods=["*"], allow_headers=["*"])

DB = None

@app.post("/connect")
async def connect(req: Request):
    global DB
    body = await req.json()
    url = body.get("url") or f"postgresql://{body['user']}:{body['pass']}@{body['host']}:{body['port']}/{body['db']}"
    DB = await asyncpg.create_pool(url)
    return {"ok": True}

@app.get("/jugadores")
async def get_jugadores():
    rows = await DB.fetch("SELECT * FROM jugadores ORDER BY id")
    return [dict(r) for r in rows]

@app.post("/jugadores")
async def add_jugador(req: Request):
    b = await req.json()
    await DB.execute("INSERT INTO jugadores (server_id,nombre,tag) VALUES ($1,$2,$3)", b["server_id"], b["nombre"], b["tag"])
    return {"ok": True}

@app.put("/jugadores/{id}")
async def update_jugador(id: int, req: Request):
    b = await req.json()
    await DB.execute("UPDATE jugadores SET nombre=$1,tag=$2,ultimo_rango=$3 WHERE id=$4", b["nombre"], b["tag"], b.get("ultimo_rango"), id)
    return {"ok": True}

@app.delete("/jugadores/{id}")
async def del_jugador(id: int):
    await DB.execute("DELETE FROM jugadores WHERE id=$1", id)
    return {"ok": True}

@app.get("/partidas")
async def get_partidas(jugador: str = "", modo: str = ""):
    q = "SELECT * FROM partidas WHERE 1=1"
    params = []
    if jugador:
        params.append(f"%{jugador}%")
        q += f" AND jugador_nombre ILIKE ${len(params)}"
    if modo and modo != "%":
        params.append(modo)
        q += f" AND modo = ${len(params)}"
    q += " ORDER BY fecha DESC LIMIT 200"
    rows = await DB.fetch(q, *params)
    return [dict(r) for r in rows]

@app.put("/partidas/{match_id}/{nombre}/{tag}")
async def update_partida(match_id: str, nombre: str, tag: str, req: Request):
    b = await req.json()
    await DB.execute(
        "UPDATE partidas SET kills=$1,deaths=$2,assists=$3,acs=$4,won=$5,mapa=$6,agente=$7,adr=$8,hs=$9 WHERE match_id=$10 AND jugador_nombre=$11 AND jugador_tag=$12",
        b["kills"], b["deaths"], b["assists"], b["acs"], b["won"], b["mapa"], b["agente"], b.get("adr"), b.get("hs"), match_id, nombre, tag)
    return {"ok": True}

@app.delete("/partidas/{match_id}/{nombre}/{tag}")
async def del_partida(match_id: str, nombre: str, tag: str):
    await DB.execute("DELETE FROM partidas WHERE match_id=$1 AND jugador_nombre=$2 AND jugador_tag=$3", match_id, nombre, tag)
    return {"ok": True}

@app.get("/stats")
async def get_stats():
    j = await DB.fetchval("SELECT COUNT(*) FROM jugadores")
    p = await DB.fetchval("SELECT COUNT(*) FROM partidas")
    w = await DB.fetchval("SELECT COUNT(*) FROM partidas WHERE won=true")
    modos = await DB.fetch("SELECT modo, COUNT(*) as c FROM partidas GROUP BY modo ORDER BY c DESC")
    agentes = await DB.fetch("SELECT agente, COUNT(*) as c FROM partidas WHERE agente!='Desconocido' GROUP BY agente ORDER BY c DESC LIMIT 5")
    return {"jugadores": j, "partidas": p, "wins": w, "modos": [dict(r) for r in modos], "top_agentes": [dict(r) for r in agentes]}

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="127.0.0.1", port=7788)
